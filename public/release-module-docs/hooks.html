useAllReleases

Query

Fetches paginated releases with filtering and sorting support.

import { useAllReleases } from '@/hooks/releases/useAllReleases';

const { data, isLoading, refetch } = useAllReleases({
  filter: { 
    status: ['active', 'uat'], 
    search: 'mobile' 
  },
  sort: { column: 'target_date', direction: 'asc' },
  page: 0,
  pageSize: 12,
});

// Returns
interface AllReleasesResult {
  releases: Release[];
  total: number;
}
ParameterTypeDescription
filterReleasesFilterStatus array and search text
sortReleasesSortColumn name and direction
pagenumberPage index (0-based)
pageSizenumberItems per page

useReleasesSelection

State

Manages multi-select state for bulk actions.

import { useReleasesSelection } from '@/hooks/releases/useReleasesSelection';

const ids = releases.map(r => r.id);
const { 
  selected, 
  toggle, 
  toggleAll, 
  clear, 
  selectAllState 
} = useReleasesSelection(ids);

// Toggle single item
toggle('release-uuid');

// Toggle all items
toggleAll();

// Clear selection
clear();

// Check selection state
console.log(selected.size); // Set
console.log(selectAllState); // 'none' | 'some' | 'all'

useCreateRelease

Mutation

Creates a new release with automatic cache invalidation.

import { useCreateRelease } from '@/hooks/useReleases';

const createRelease = useCreateRelease();

// Create release
createRelease.mutate({
  release_vehicle_id: 'vehicle-uuid',
  name: 'Release 2.0',
  target_date: '2026-03-01',
  notes: 'Major feature release',
});

// With callbacks
createRelease.mutate(data, {
  onSuccess: (release) => {
    console.log('Created:', release.id);
  },
  onError: (error) => {
    console.error(error.message);
  },
});
Input FieldTypeRequired
release_vehicle_idstringYes
namestringYes
target_datestringNo
notesstringNo
