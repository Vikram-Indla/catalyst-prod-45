releases

releases

Core release entity tracking software releases

Column Type Constraints Description
idUUIDPKPrimary key
nameTEXTNOT NULLRelease name
versionTEXTDEFAULT 'v1.0'Version identifier
descriptionTEXTnullableRelease description
statusTEXTDEFAULT 'planned'planned|active|uat|shipped|cancelled
target_dateTIMESTAMPTZnullableTarget release date
release_dateTIMESTAMPTZnullableActual release date
release_vehicle_idUUIDFK→ release_vehicles
owner_idUUIDFK→ profiles
readiness_pctNUMERIC(5,2)nullableReadiness percentage
notesTEXTnullableAdditional notes
created_atTIMESTAMPTZDEFAULT now()Created timestamp
updated_atTIMESTAMPTZDEFAULT now()Updated timestamp
 
CREATE TABLE releases (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL,
  version TEXT DEFAULT 'v1.0',
  description TEXT,
  status TEXT DEFAULT 'planned',
  target_date TIMESTAMP WITH TIME ZONE,
  release_date TIMESTAMP WITH TIME ZONE,
  release_vehicle_id UUID REFERENCES release_vehicles(id),
  owner_id UUID REFERENCES profiles(id),
  readiness_pct NUMERIC(5,2),
  notes TEXT,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT now(),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT now()
);

CREATE TABLE defects (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  defect_id TEXT NOT NULL UNIQUE,
  defect_key TEXT,
  title TEXT NOT NULL,
  description TEXT,
  severity TEXT NOT NULL,
  priority TEXT NOT NULL,
  status TEXT DEFAULT 'open',

  environment TEXT,
